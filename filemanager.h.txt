#ifndef FILEMANAGER_H
#define FILEMANAGER_H
#include <fstream>
#include <iostream>
using namespace std;

class FileManager{
public:
    void save(string data){
        ofstream out("log.txt",ios::app);
        out<<data<<endl;
        out.close();
    }

    void show(){
        ifstream in("log.txt");
        string s;
        while(getline(in,s)){
            cout<<s<<endl;
        }
        in.close();
    }
};

#endif
